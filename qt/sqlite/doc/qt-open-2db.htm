qt中连接两个数据库

#include "mainwindow.h"
#include <QApplication>
#include <QSqlDatabase>
#include <QDebug>
#include <QSqlDatabase>
#include <QSqlQuery>

int main(int argc, char *argv[])
{
    QApplication a(argc, argv);

    QSqlDatabase dbprofile = QSqlDatabase::addDatabase("QSQLITE", "profile_connect");
    dbprofile.setDatabaseName("../profile.db");

    if(!dbprofile.open())
        qDebug() << "profilesconn no";
    else
        qDebug() << "profilesconn open ok!";

    QSqlDatabase dbdata= QSqlDatabase::addDatabase("QSQLITE", "data_connect");
    dbdata.setDatabaseName("../data.db");
    if(!dbdata.open())
        qDebug() << "data no";
    else
        qDebug() << "data open ok!";


    QSqlQuery query(dbdata);
    qDebug() << query.exec("select * from profile");
    query.next();
    qDebug() << query.value(0).toString();

    QSqlQuery query1(dbprofile);
    qDebug() << query1.exec("select * from profile");
    query1.next();
    qDebug() << query1.value(0).toString();

    MainWindow *w;
    w = new MainWindow;
    w->profiledb(&dbprofile);
    w->datadb(&dbdata);
    w->show();

    return a.exec();
}
